def getFileName(p):

    fileall = [];

        onlyfiles = [f for f in onlyfiles if ".jpg" in f]

    onlyfile_list = [y for x in onlylist for y in x]
    onlyfile_list = [x.strip('.jpg') for x in onlyfile_list]

    return (flattened_list, onlyfile_list)
